객체 조작 메소드

종류 사용법 설명
html() html('요소') $('요소 선택').html();
$('요소 선택').html('요소');		 하위 요소를 가져온다.
하위 요소를 제거하고, 새 요소를 생성.
text('')
text('blabla')		 $('요소 선택').text();
$('요소 선택').text('blabla');		 텍스트를 가져온다.
하위 요소를 제거하고, 전달된 텍스트를 생성.
attr('속성')
attr('속성', '값')		 $('요소 선택').attr('속성')
$('요소 선택').attr('속성', '값')		 해당 속성값을 가져옴
해당 속성값을 교체
val()
val('값')		 $('요소 선택').val()
$('요소 선택').val('값')		 폼 요소'의 value를 가져옴
'폼 요소'의 value 속성에 주어진 값을 적용
css('속성')
css('속성', '값')		 $('요소 선택').css('속성')
$('요소 선택').css('속성', '값')		 주어진 css 속성값을 가져옴
주어진 css 속성값으로 교체
addClass('class값') $('요소 선택').addClass('class값') class 속성에서 주어진 값 추가
removeClass('class값') $('요소 선택').removeClass('class값') class 속성에서 주어진 값 제거

appendTo() $('요소').appendTo('요소 선택') 선택한 요소 내 마지막에 요소 추가
prepend() $('요소 선택').prepend('요소') 선택한 요소 내 맨 앞에 요소 추가
prependTo() $('요소').prependTo('요소 선택') 선택한 요소 내 맨 앞에 요소 추가
empty() $('요소 선택').empty() 선택한 요소의 하위 요소를 모두 삭제
remove() $('요소 선택').remove() 선택한 요소를 삭제
